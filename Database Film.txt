CREATE DATABASE data_film;
USE data_film;

CREATE TABLE film (
	id integer PRIMARY KEY AUTO_INCREMENT,
    title VARCHAR(255) NOT NULL,
    release_year integer NOT NULL,
    rating float NOT NULL DEFAULT 0.0
);
